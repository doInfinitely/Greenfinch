/**
 * Property Type Classification Function
 * 
 * Classifies properties into four categories based on usedesc and zoning_description:
 * - "single_family" (includes SFR, duplex, triplex, quadplex, townhouse, patio homes)
 * - "multifamily" (apartments, MF districts, 5+ units)
 * - "commercial" (retail, office, industrial, business)
 * - "public" (government, parks, schools, institutional, civic, agricultural)
 * 
 * Returns null if classification cannot be determined.
 */

export type PropertyClassification = 
  | "single_family"   // SFR, duplex, triplex, quadplex, townhouse, patio home
  | "multifamily"     // Apartments, 5+ units
  | "commercial"      // Retail, office, industrial, mixed-use commercial
  | "public"          // Government, parks, schools, institutional
  | null;             // Unable to classify

export interface ClassificationResult {
  classification: PropertyClassification;
  confidence: "high" | "medium" | "low";
  reason: string;
}

/**
 * Classifies a property based on usedesc and zoning_description
 */
export function classifyPropertyType(
  usedesc: string | null | undefined,
  zoningDescription: string | null | undefined
): ClassificationResult {
  const use = (usedesc || "").toLowerCase().trim();
  const zoning = (zoningDescription || "").toLowerCase().trim();
  
  if (!use && !zoning) {
    return { classification: null, confidence: "low", reason: "No usedesc or zoning_description provided" };
  }

  // STEP 1: Check zoning_description first (more specific)
  
  if (matchesPublicInstitutional(zoning)) {
    return { classification: "public", confidence: "high", reason: `Zoning indicates public/institutional: "${zoningDescription}"` };
  }
  
  if (matchesSingleFamily(zoning)) {
    return { classification: "single_family", confidence: "high", reason: `Zoning indicates single-family: "${zoningDescription}"` };
  }
  
  if (matchesMultifamily(zoning)) {
    return { classification: "multifamily", confidence: "high", reason: `Zoning indicates multifamily: "${zoningDescription}"` };
  }
  
  if (matchesCommercial(zoning)) {
    return { classification: "commercial", confidence: "high", reason: `Zoning indicates commercial: "${zoningDescription}"` };
  }
  
  // STEP 2: Fall back to usedesc
  
  if (use === "commercial") {
    return { classification: "commercial", confidence: "medium", reason: `usedesc is "Commercial"` };
  }
  
  if (use === "bpp") {
    return { classification: "commercial", confidence: "medium", reason: `usedesc is "BPP" (Business Personal Property)` };
  }
  
  if (use === "residential") {
    if (/mf|apartment|multi/.test(zoning)) {
      return { classification: "multifamily", confidence: "medium", reason: `usedesc "Residential" with MF zoning hints` };
    }
    return { classification: "single_family", confidence: "low", reason: `usedesc "Residential" - defaulting to single_family` };
  }
  
  // STEP 3: Handle special permits (S-P-1, S-P-2, PUD)
  
  if (/s-p|pud|planned development/.test(zoning)) {
    if (/r-mf|m-f|\bmf\b|apartment|multi/.test(zoning)) {
      return { classification: "multifamily", confidence: "medium", reason: `Special permit with MF indicators` };
    }
    if (/single family|\bsf\b|r-6|r-7|r-10|duplex|residential/.test(zoning)) {
      return { classification: "single_family", confidence: "medium", reason: `Special permit with SF indicators` };
    }
    if (/commercial|retail|office|industrial|fwy|hotel|auto/.test(zoning)) {
      return { classification: "commercial", confidence: "medium", reason: `Special permit with commercial indicators` };
    }
  }
  
  return { classification: null, confidence: "low", reason: `Unable to classify` };
}

// =============================================================================
// PATTERN MATCHING FUNCTIONS
// =============================================================================

function matchesSingleFamily(zoning: string): boolean {
  // Explicit exclusions - if these are present, it's NOT single family
  if (/apartment|multi[- ]?family|\bmf\b|multiple family/i.test(zoning)) {
    return false;
  }
  
  const patterns = [
    // Explicit single family
    /single[- ]?family/i,
    /\bsf[-\s]?\d/i,          // SF-7, SF-10, etc.
    /\bsf[-\s]?e\b/i,         // SF-E (estate)
    /one family/i,
    
    // Duplex, triplex, quadplex (1-4 units)
    /duplex/i,
    /triplex/i,
    /quadplex/i,
    /two[- ]?family/i,
    /2[- ]?family/i,
    /\br-6\b/i,
    /\br-7\b/i,
    /\br-10\b/i,
    /\br-40\b/i,
    
    // Townhouse / Patio homes
    /townhouse/i,
    /townhome/i,
    /\bth\b/i,
    /patio home/i,
    /zero lot line/i,
    /attached residential/i,
    
    // Estate / Low density
    /estate district/i,
    /country estate/i,
    /low density/i,
    /medium density/i,
    
    // Residential districts with numbers
    /residential district \d/i,
    /residence district \d/i,
    
    // Manufactured/mobile homes (not parks)
    /manufactured home(?! park)/i,
    /mobile home(?! park)/i,
    
    // Other SF patterns
    /clustered housing/i,
    /unified residential/i,
    /walkable urban residential/i,
  ];
  
  return patterns.some(p => p.test(zoning));
}

function matchesMultifamily(zoning: string): boolean {
  const patterns = [
    /apartment/i,
    /multi[- ]?family/i,
    /multiple family/i,
    /\bmf\b/i,
    /\bmf[-\s]?\d/i,
    /\br-mf/i,
    /mobile home park/i,
    /senior living/i,
    /assisted living/i,
    /tod for mf/i,
    /mixed use residential/i,   // Residential-focused mixed use
  ];
  
  return patterns.some(p => p.test(zoning));
}

function matchesCommercial(zoning: string): boolean {
  // Exclude residential-focused mixed use
  if (/mixed use residential|residential mixed use/i.test(zoning)) {
    return false;
  }
  
  const patterns = [
    // Retail
    /retail/i,
    /shopping/i,
    /commercial/i,
    /business district/i,
    /business park/i,
    /downtown/i,
    /central business/i,
    /neighborhood service/i,
    /community service/i,
    
    // Office
    /\boffice\b/i,
    /\bc-ou/i,
    
    // Industrial / Manufacturing
    /industrial/i,
    /manufacturing/i,
    /warehouse/i,
    /logistics/i,
    /\blr\b/i,
    /\bip\b/i,
    /\bns\b/i,
    /port industrial/i,
    
    // Mixed-use (commercial-focused)
    /mixed use(?! residential)/i,
    /urban center/i,
    /town center/i,
    /turnpike/i,
    /freeway district/i,
    /highway commercial/i,
    /belt line/i,
    /transit center/i,
    /old town/i,
    
    // Specific commercial
    /hotel/i,
    /service station/i,
    /\bauto\b/i,
  ];
  
  return patterns.some(p => p.test(zoning));
}

function matchesPublicInstitutional(zoning: string): boolean {
  const patterns = [
    // Government
    /government/i,
    /governmental/i,
    /city park/i,
    /city property/i,
    /civic/i,
    
    // Parks / Open space
    /existing parks/i,
    /open space/i,
    /conservation/i,
    /floodplain(?! residential)/i,
    
    // Education
    /school/i,
    /university/i,
    /i\.s\.d\./i,
    /campus district/i,
    
    // Religious / Institutional
    /institutional/i,
    /religious/i,
    /church/i,
    
    // Agricultural
    /agricultural/i,
    /agriculture/i,
    
    // Utilities
    /transmission line/i,
    /easement/i,
    /take line/i,
  ];
  
  return patterns.some(p => p.test(zoning));
}

// =============================================================================
// SIMPLIFIED HELPER
// =============================================================================

/**
 * Simplified version - just returns the classification string or null
 */
export function getPropertyType(
  usedesc: string | null | undefined,
  zoningDescription: string | null | undefined
): PropertyClassification {
  return classifyPropertyType(usedesc, zoningDescription).classification;
}
